class Solution {
    public int findJudge(int n, int[][] trust) {
        if(n==1){
            return 1 ;
        }
        int inBound[] = new int[n+1] ;
        int outBound[] = new int[n+1] ;

        for(int p[]: trust ){
           int a = p[0];
           int b = p[1];
           outBound[a]++;
           inBound[b]++;
        }
        
        for(int i=0 ;i <n+1 ;i++){
            if(inBound[i]==n-1 && outBound[i]==0){
                return i ;
            }
        }

        return -1 ;

    }
}
